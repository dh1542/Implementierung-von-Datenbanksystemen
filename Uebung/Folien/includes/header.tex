%c from texinfo.tex
\def\ifmonospace{\ifdim\fontdimen3\font=0pt }

%c C plus plus
\def\C++{%
\ifmonospace%
    C++%
\else%
    C\kern-.1em\raise.30ex\hbox{\smaller{++}}%
\fi%
\spacefactor1000 }

%c C sharp
\def\Csharp{%
\ifmonospace%
    \Csharp%
\else%
    C\kern-.1em\raise.30ex\hbox{\smaller{\#}}%
\fi%
\spacefactor1000 }

%c Cool Quote
\newcommand{\quoteframe}[3]
{
	\begin{frame}
		\begin{quote}
		\Huge#3
		\end{quote}
		\vskip0pt plus 1filll
		\begin{minipage}{\textwidth}
			\begin{flushright}
			\LARGE#1
			\\
			\normalsize#2
			\end{flushright}
		\end{minipage}
	\end{frame}
}
