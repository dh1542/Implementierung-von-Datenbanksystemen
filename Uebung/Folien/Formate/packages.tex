\newcommand{\quoteframe}[3]
{{
\setbeamertemplate{logo}{}
	\begin{frame}
	\begin{quote}
		\Huge#1
	\end{quote}
	\vskip0pt plus 1filll
	\begin{minipage}{\textwidth}
		\begin{flushright}
			{\LARGE#2 }
			\\
			{\normalsize#3 }
		\end{flushright}
	\end{minipage}
\end{frame}
}}

%picture, author, source
\newcommand{\pictureframe}[3]
{{
\setbeamertemplate{logo}{}
	\begin{frame}
	#1
	\vskip0pt plus 1filll
	\begin{minipage}{\textwidth}
		\begin{flushright}
			{\LARGE#2\hspace{70pt} }
			\\
			{\normalsize#3\hspace{70pt} }
		\end{flushright}
	\end{minipage}
\end{frame}
}}
\newcommand{\fullpictureframe}[2][]{
\begin{frame}[c]{#1}
\begin{center}
	\includegraphics[width=\textwidth, height=\textheight, keepaspectratio]{#2}
\end{center}
\end{frame}
}
